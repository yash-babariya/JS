const stringFormatter = {
    capitalize: str => 
        str.charAt(0).toUpperCase() + str.slice(1).toLowerCase(),
    camelCase: str => 
        str.toLowerCase().replace(/[^a-zA-Z0-9]+(.)/g, 
            (m, chr) => chr.toUpperCase()),
    slug: str => 
        str.toLowerCase().trim()
           .replace(/[^\w\s-]/g, '')
           .replace(/[\s_-]+/g, '-'),
    truncate: (str, len) => 
        str.length > len ? str.slice(0, len) + '...' : str
};

ЁЯОп ркХрк╕рк░ркд 2: ркЯрлЗркорлНрккрк▓рлЗркЯ ркПркирлНркЬрк┐рки

const templateEngine = {
    render(template, data) {
        return template.replace(/\${(.*?)}/g, 
            (match, key) => {
                return data[key.trim()] || '';
            });
    },
    compile(template) {
        return data => this.render(template, data);
    }
};

const stringValidator = {
    isEmail: str => 
        /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(str),
    isURL: str => 
        /^(https?:\/\/)?([\da-z\.-]+)\.([a-z\.]{2,6})([\/\w \.-]*)*\/?$/.test(str),
    isStrongPassword: str => 
        /^(?=.*\d)(?=.*[a-z])(?=.*[A-Z])(?=.*[!@#$%^&*]).{8,}$/.test(str),
    containsOnly: (str, allowed) => 
        new RegExp(`^[${allowed}]+$`).test(str)
};
