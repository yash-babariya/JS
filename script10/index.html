const player = document.querySelector('.player');
let position = { x: 0, y: 0 };

document.addEventListener('keydown', function(e) {
    switch(e.key) {
        case 'ArrowUp':
            position.y -= 10;
            break;
        case 'ArrowDown':
            position.y += 10;
            break;
        case 'ArrowLeft':
            position.x -= 10;
            break;
        case 'ArrowRight':
            position.x += 10;
            break;
    }
    player.style.transform = 
        `translate(${position.x}px, ${position.y}px)`;
});
