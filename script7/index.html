એરે મેથડ્સ રેફરન્સ

ઉદાહરણો સાથે એરે મેથડ્સની સંપૂર્ણ યાદી

// Test array
const arr = [1, 2, 3, 4, 5];
const fruits = ['apple', 'banana', 'orange'];

// Adding/Removing Elements
arr.push(6);                  // [1,2,3,4,5,6]
arr.pop();                    // [1,2,3,4,5] returns: 6
arr.unshift(0);              // [0,1,2,3,4,5]
arr.shift();                 // [1,2,3,4,5] returns: 0

// Modifying Arrays
arr.splice(2, 1);            // [1,2,4,5] removes 1 element at index 2
arr.splice(2, 0, 3);         // [1,2,3,4,5] adds element at index 2
arr.slice(1, 3);             // [2,3] returns new array
arr.concat([6, 7]);          // [1,2,3,4,5,6,7] returns new array

// Array Transformations
arr.map(x => x * 2);         // [2,4,6,8,10]
arr.filter(x => x > 3);      // [4,5]
arr.reduce((a,b) => a + b);  // 15 (sum of all elements)
arr.every(x => x > 0);       // true (all elements > 0)
arr.some(x => x > 4);        // true (at least one element > 4)
arr.find(x => x > 3);        // 4 (first element > 3)
arr.findIndex(x => x > 3);   // 3 (index of first element > 3)

// Array Order Operations
arr.reverse();               // [5,4,3,2,1]
arr.sort();                  // [1,2,3,4,5]
arr.sort((a,b) => b - a);    // [5,4,3,2,1] (descending)

// Array Information
arr.includes(3);             // true
arr.indexOf(3);              // 2
arr.lastIndexOf(3);          // 2
arr.join(', ');              // "1, 2, 3, 4, 5"

// ES2019+ Methods
arr.flat();                  // Flattens nested arrays
arr.flatMap(x => [x, x*2]);  // [1,2,2,4,3,6,4,8,5,10]

// Iterating Methods
arr.forEach(x => console.log(x)); // Logs: 1,2,3,4,5
arr.entries();               // Iterator [index, element]
arr.keys();                  // Iterator of indices
arr.values();                // Iterator of elements

// Array Type Checking
Array.isArray(arr);          // true
arr.constructor === Array;   // true

// Creating Arrays
Array.from('123');          // ['1','2','3']
Array.of(1, 2, 3);         // [1,2,3]
new Array(3);              // [empty × 3]
Array(3).fill(0);          // [0,0,0]

// Advanced Operations
const unique = [...new Set(arr)];  // Remove duplicates
const max = Math.max(...arr);      // Find maximum
const min = Math.min(...arr);      // Find minimum

// Chaining Methods
arr.filter(x => x > 2)
   .map(x => x * 2)
   .reduce((a,b) => a + b);  // 24 (sum of doubled numbers > 2)

// Performance Tips
// 1. Use typed arrays for numeric data
const typedArr = new Int32Array([1,2,3]);

// 2. Pre-allocate array length when possible
const preAllocated = new Array(1000);

// 3. Use appropriate method for searching
// - indexOf for small arrays
// - Set for large unique values
// - Object lookup for key-value pairs

const arrayUtils = {
    toTree: (arr, id = 'id', parentId = 'parentId') => {
        const nodes = {};
        return arr.filter(item => {
            nodes[item[id]] = { ...item, children: [] };
            const parent = nodes[item[parentId]];
            parent ? parent.children.push(nodes[item[id]]) : null;
            return !item[parentId];
        });
    },
    chunk: (arr, size) => {
        return Array.from({ length: Math.ceil(arr.length / size) }, 
            (_, i) => arr.slice(i * size, i * size + size));
    }
};

    findElement: (arr, element) => arr.indexOf(element),
    countOccurrences: (arr, element) => 
        arr.reduce((count, current) => 
            current === element ? count + 1 : count, 0),
    removeDuplicates: arr => [...new Set(arr)],
    removeElement: (arr, element) => 
        arr.filter(item => item !== element)
};

const advancedArray = {
    flattenArray: arr => arr.reduce((flat, item) => 
        flat.concat(Array.isArray(item) ? 
            advancedArray.flattenArray(item) : item), []),
            
    chunkArray: (arr, size) => 
        Array.from({ length: Math.ceil(arr.length / size) },
            (_, i) => arr.slice(i * size, i * size + size)),
            
    sortByProperty: (arr, prop) => 
        [...arr].sort((a, b) => 
            typeof a[prop] === 'string' ? 
                a[prop].localeCompare(b[prop]) : 
                a[prop] - b[prop]),
                
    findDuplicates: arr => 
        arr.filter((item, index) => 
            arr.indexOf(item) !== index)
};

🎯 કસરત 4: પ્રેક્ટિકલ એપ્લિકેશન્સ

ડેટા પ્રોસેસિંગ અને એનાલિસિસ માટે રીયલ-વર્લ્ડ એરે યુટિલિટીઝ બનાવો

const arrayAnalytics = {
    statistics: arr => ({
        average: arr.reduce((sum, num) => sum + num, 0) / arr.length,
        median: arr => {
            const sorted = [...arr].sort((a, b) => a - b);
            const mid = Math.floor(sorted.length / 2);
            return sorted.length % 2 === 0 ? 
                (sorted[mid - 1] + sorted[mid]) / 2 : 
                sorted[mid];
        },
        mode: arr => {
            const frequency = {};
            arr.forEach(item => frequency[item] = (frequency[item] || 0) + 1);
            return Object.entries(frequency)
                .reduce((a, b) => a[1] > b[1] ? a : b)[0];
        }
    }),

    groupAndCount: (arr, keyFn) => 
        arr.reduce((groups, item) => {
            const key = keyFn(item);
            groups[key] = (groups[key] || 0) + 1;
            return groups;
        }, {}),

    pivotTable: (arr, rowKey, colKey, valueFn) => 
        arr.reduce((table, item) => {
            table[item[rowKey]] = table[item[rowKey]] || {};
            table[item[rowKey]][item[colKey]] = valueFn(item);
            return table;
        }, {})
};
