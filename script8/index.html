Object Basics

Creating and manipulating objects

// Object creation
const person = {
    name: 'John',
    age: 30,
    greet() {
        return `Hello, I'm ${this.name}`;
    }
};

// Object methods
Object.keys(person);      // Get keys
Object.values(person);    // Get values
Object.entries(person);   // Get key-value pairs
Object.freeze(person);    // Make immutable

Object Methods Reference

Complete list of object methods with examples

// Test object
const obj = { name: 'John', age: 30 };
const nested = { user: { details: { id: 1 } } };

// Object Creation Methods
Object.create(null);                // Create object with no prototype
Object.assign({}, obj);             // Clone object
Object.fromEntries([['a', 1]]);     // Create from key-value pairs

// Property Access Methods
Object.keys(obj);                   // ['name', 'age']
Object.values(obj);                 // ['John', 30]
Object.entries(obj);                // [['name','John'], ['age',30]]
Object.getOwnPropertyNames(obj);    // All properties including non-enumerable
Object.getOwnPropertySymbols(obj);  // Symbol properties

// Property Descriptors
Object.defineProperty(obj, 'id', {
    value: 1,
    writable: false,               // Cannot change value
    enumerable: true,              // Shows up in loops
    configurable: false            // Cannot delete or modify descriptor
});

Object.defineProperties(obj, {
    prop1: { value: 1, writable: true },
    prop2: { value: 2, writable: false }
});

Object.getOwnPropertyDescriptor(obj, 'name');   // Get single descriptor
Object.getOwnPropertyDescriptors(obj);          // Get all descriptors

// Object State Methods
Object.freeze(obj);                // Make immutable
Object.seal(obj);                  // Prevent adding/deleting properties
Object.preventExtensions(obj);     // Prevent adding properties

Object.isFrozen(obj);              // Check if frozen
Object.isSealed(obj);              // Check if sealed
Object.isExtensible(obj);          // Check if extensible

// Prototype Methods
Object.getPrototypeOf(obj);        // Get prototype
Object.setPrototypeOf(obj, null);  // Set prototype

// Modern Object Methods (ES2020+)
Object.hasOwn(obj, 'name');        // true (safer than hasOwnProperty)
Object.is(NaN, NaN);               // true (strict equality)

// Common Patterns
// 1. Deep Clone
const deepClone = obj => JSON.parse(JSON.stringify(obj));

// 2. Merge Objects
const merged = { ...obj1, ...obj2 };

// 3. Optional Chaining
const value = nested?.user?.details?.id;  // Safe nested access

// 4. Nullish Coalescing
const name = obj.name ?? 'Anonymous';     // Default value

// 5. Dynamic Properties
const key = 'dynamicKey';
const dynamic = { [key]: 'value' };

// 6. Property Shorthand
const name = 'John';
const shorthand = { name };              // Same as { name: name }

// 7. Method Shorthand
const methods = {
    sayHi() { return 'Hi!' }            // Instead of sayHi: function() {}
};

// 8. Computed Properties
const prefix = 'user';
const computed = {
    [`${prefix}_id`]: 1,           // Dynamic property name
    [`${prefix}_name`]: 'John'
};

// Performance Tips
// 1. Use Object.create(null) for pure dictionaries
// 2. Avoid delete operator when possible
// 3. Use Map for frequent additions/removals
// 4. Use WeakMap for object keys
// 5. Batch property updates

Advanced Object Concepts

Prototypes, inheritance, and descriptors

// Prototypal inheritance
class Animal {
    constructor(name) {
        this.name = name;
    }
    speak() {
        return `${this.name} makes a sound`;
    }
}

class Dog extends Animal {
    speak() {
        return `${this.name} barks`;
    }
}

// Object descriptors
Object.defineProperty(person, 'id', {
    value: 1,
    writable: false,
    enumerable: true,
    configurable: false
});

Object Patterns

Common object design patterns

// Factory pattern
const createUser = (name, role) => ({
    name,
    role,
    permissions: getPermissions(role),
    timestamp: new Date()
});

// Singleton pattern
const Database = (() => {
    let instance;
    
    return {
        getInstance() {
            if (!instance) {
                instance = {
                    connect() { /* ... */ },
                    query() { /* ... */ }
                };
            }
            return instance;
        }
    };
})();

🎯 Exercise 1: Object Manipulation

Create utility functions for common object operations

const objectUtils = {
    deepClone: obj => JSON.parse(JSON.stringify(obj)),
    
    merge: (target, ...sources) => 
        sources.reduce((acc, src) => {
            Object.keys(src).forEach(key => {
                acc[key] = src[key];
            });
            return acc;
        }, target),
    
    flatten: (obj, prefix = '') =>
        Object.keys(obj).reduce((acc, key) => {
            const prop = prefix ? `${prefix}.${key}` : key;
            if (typeof obj[key] === 'object') {
                Object.assign(acc, flatten(obj[key], prop));
            } else {
                acc[prop] = obj[key];
            }
            return acc;
        }, {})
};

🎯 Exercise 2: Object-Oriented Design

Implement a class hierarchy with inheritance

class Vehicle {
    constructor(make, model, year) {
        this.make = make;
        this.model = model;
        this.year = year;
    }

    getInfo() {
        return `${this.year} ${this.make} ${this.model}`;
    }
}

class Car extends Vehicle {
    constructor(make, model, year, doors) {
        super(make, model, year);
        this.doors = doors;
        this.type = 'Car';
    }
}

class Motorcycle extends Vehicle {
    constructor(make, model, year, type) {
        super(make, model, year);
        this.type = type;
    }
}

🎯 Exercise 3: Advanced Patterns

Implement advanced object patterns and state management

const StateManager = {
    state: {},
    listeners: [],

    setState(newState) {
        this.state = { ...this.state, ...newState };
        this.notify();
    },

    getState() {
        return { ...this.state };
    },

    subscribe(listener) {
        this.listeners.push(listener);
        return () => {
            this.listeners = 
                this.listeners.filter(l => l !== listener);
        };
    },

    notify() {
        this.listeners.forEach(listener => 
            listener(this.state));
    }
};
