const userPreferences = {
    save(preferences) {
        localStorage.setItem('userPrefs', 
            JSON.stringify(preferences));
    },
    load() {
        const prefs = localStorage.getItem('userPrefs');
        return prefs ? JSON.parse(prefs) : {};
    },
    update(key, value) {
        const prefs = this.load();
        prefs[key] = value;
        this.save(prefs);
    }
};
